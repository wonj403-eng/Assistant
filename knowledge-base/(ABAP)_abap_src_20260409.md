SELECT IDNO AS EMPNO, 
    NVL(DISPDEPTCD, DEPTCD) AS DEPTCD,
    NVL(DISPDETPCD, DEPTNM) AS DEPTNM
FROM PDSS40.HRI_ORD_MAS A
WHERE 1=1
AND A.STATUS IN ('C','U')
AND A.APPNTDT = (SELECT MAX(APPNTDT) FROM PDSS40.HRI_ORD_MAS B
                 WHERE 1=1 AND B.IDNO = A.IDNO)
